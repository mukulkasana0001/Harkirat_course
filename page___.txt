import { paginationContainer, paginationInFO } from "./common.js";
import { updateTableUI } from "./updateTableUI.js";



// Function to display data for the current page
function displayPage(page, DataTOSeprate) {
    paginationInFO.currentPage = page;
    const startIndex = (paginationInFO.currentPage - 1) * paginationInFO.itemsPerPage;
    const endIndex = startIndex + paginationInFO.itemsPerPage;
    const currentItems = DataTOSeprate.slice(startIndex, endIndex);

    updateTableUI(currentItems)
}

// Function to set up pagination buttons
function setupPagination(DataTOSeprate) {
    paginationContainer.innerHTML = ''; // Clear current pagination
    const pageCount = Math.ceil(DataTOSeprate.length / paginationInFO.itemsPerPage);

    // Previous button
    const prevButton = document.createElement('button');
    prevButton.textContent = 'Previous';
    prevButton.className = `px-3 py-2 mx-1 rounded-md text-sm font-medium ${paginationInFO.currentPage === 1 ? 'opacity-50 cursor-not-allowed' : 'bg-blue-500 text-white hover:bg-blue-600'
        }`;
    prevButton.addEventListener('click', () => {
        if (paginationInFO.currentPage > 1) {
            displayPage(paginationInFO.currentPage - 1, DataTOSeprate);
            setupPagination(DataTOSeprate);
        }
    });
    paginationContainer.appendChild(prevButton);


    //     Adding of
    const pageno = document.createElement('button');
    pageno.textContent = ` page ${pageCount} of`;
    paginationContainer.appendChild(pageno);

    // Page numbers
    for (let i = paginationInFO.currentPage; i <= paginationInFO.currentPage && i <= pageCount; i++) {
        const button = document.createElement('button');
        button.textContent = i;
        button.className = `px-3 py-2 mx-1 rounded-md text-sm font-medium ${i === paginationInFO.currentPage ? 'bg-blue-600 text-white' : 'bg-gray-200 text-gray-700 hover:bg-gray-300'
            }`;
        button.addEventListener('click', () => {
            displayPage(i, DataTOSeprate);
            setupPagination(DataTOSeprate);
        });
        paginationContainer.appendChild(button);
    }

    // Next button
    const nextButton = document.createElement('button');
    nextButton.textContent = 'Next';
    nextButton.className = `px-3 py-2 mx-1 rounded-md text-sm font-medium ${paginationInFO.currentPage === pageCount ? 'opacity-50 cursor-not-allowed' : 'bg-blue-500 text-white hover:bg-blue-600'
        }`;

    nextButton.addEventListener('click', () => {
        if (paginationInFO.currentPage < pageCount) {
            displayPage(paginationInFO.currentPage + 1, DataTOSeprate);
            setupPagination(DataTOSeprate);
        }
    });
    paginationContainer.appendChild(nextButton);
}



export { displayPage, setupPagination }
